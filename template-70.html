Row Selection

The Row Selection Extension will add the ability to select a single row or multiple rows on Grid. 

efx-grid {
    height: 169px;
}

<efx-grid id="grid"></efx-grid>

await (await import("./theme-loader.js")).default; // This line is only required for demo purpose. It is not relevant for using EFX Grid in your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
var records = DataGenerator.generateRecords(fields, { numRows: 5 });
var configObj = {
    sorting: {
        sortableColumns: true
    },
    columns: [
        { title: "Company", field: fields[0] },
        { title: "Market", field: fields[1], width: 120 },
        { title: "Last", field: fields[2], width: 100 },
        { title: "Net. Chng", field: fields[3], width: 100 },
        { title: "Industry", field: fields[4] }
    ],
    staticDataRows: records,
    extensions: [
        new RowSelection()
    ]
};

var grid = document.getElementById("grid");
grid.config = configObj;

basedOnContent mode

Normally, when content is sorted and a row changes position, row selection will not change position like the content does (try example above). But if you do want the row selection to relocate itself with the content, you can turn on basedOnContent mode.

For actual result, try to select one or more rows, then click the column header to sort data and see what happens with row selection.

    height: 169px;

You can directly hook native copy event on Grid element to retrieve data and put it on system clipboard. Use the extension's APIs to get currently selected rows.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
function onCopy(e) {
    var selectedRowIndices = rowSelExt.getSelectedRows();
    var text = "No Selected Row";
    if(selectedRowIndices) {
        var rows = grid.api.getMultipleRowData(selectedRowIndices);
        var values = rows.map(function getValueInTSV(row) {
            return [
                row["companyName"], 
                row["market"], 
                row["CF_LAST"], 
                row["CF_NETCHNG"], 
                row["industry"]
            ].join("\t");
        });
        text = values.join("\n");
    }
    e.clipboardData.setData("text/plain", text);
    e.preventDefault();
}

var rowSelExt = new RowSelection();
var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
var records = DataGenerator.generateRecords(fields, { numRows: 5 });
var configObj = {
    sorting: {
        sortableColumns: true
    },
    columns: [
        { title: "Company", field: fields[0] },
        { title: "Market", field: fields[1], width: 120 },
        { title: "Last", field: fields[2], width: 100 },
        { title: "Net. Chng", field: fields[3], width: 100 },
        { title: "Industry", field: fields[4] }
    ],
    staticDataRows: records,
    extensions: [
        rowSelExt
    ],
    rowSelection: {
        basedOnContent: true
    }
};

var grid = document.getElementById("grid");
grid.config = configObj;
grid.addEventListener("copy", onCopy);

Events

event

beforeSelection

Fired before selectionChanged. We can cancel the select operation by setting `cancel` property on the event arugment to true
Note that selection by JavaScript APIs cannot be prevented using this event.
Example:
var ext = new RowSelectionPlugin();
ext.listen("beforeSelection" function(e) {
  e.cancel = true; // Cancel user selection
});
event

selectionChanged

Fired when selection is changed by mouse down or key press. A change by JavaScript APIs will not trigger this event.
