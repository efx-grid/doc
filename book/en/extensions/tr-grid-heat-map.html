Heat Map

The Heat Map Extension provides heat map coloring for column fields, which will be the type set by the heatMap property.

For contributions, check out the Heat Map Extension repository.

  • HTML
  • JS
  • CSS
<emerald-grid id="grid">

</emerald-grid>
tr.DataGenerator.addFieldInfo("volume", {
  type: "number",
  min: "1",
  max: "1001"
});
var fields = ["companyName", "market", "CF_LAST", "volume", "PCTCHNG2"];
var records = tr.DataGenerator.generateRecords(fields, { numRows: 20 });
var configObj = {
  rowHighlight: true,
  rowSelection: true,
  columns: [{
      title: "Company",
      field: fields[0]
    },
    {
      title: "Market",
      field: fields[1],
      width: 120
    },
    {
      title: "Last",
      field: fields[2],
      width: 100,
      heatMap: {
        mode: "text",
        midPoint: 500
      }
    },
    {
      title: "Volume",
      field: fields[3],
      sortBy: fields[3],
      width: 100,
      heatMap: {
        midPoint: 300
      }
    },
    {

      title: "Pct. Chng",
      field: fields[4],
      sortBy: fields[4],
      width: 100,
      heatMap: true
    }
  ],
  dataModel: {
    data: records
  },
  extensions: [
    new tr.HeatMapExtension()
  ]
};

var grid = document.getElementById("grid");
grid.config = configObj;
emerald-grid {
  height: 200px;
}
