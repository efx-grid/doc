Conditional Coloring

The Conditional Coloring Extension provides a conditional coloring for a column's field, which will be triggered when a condition is met. For example, fill green when the value is more than zero.

For contributions, check out the Conditional Coloring Extension repository.

  • HTML
  • JS
  • CSS
<emerald-grid id="grid">

</emerald-grid>
var configObj = {
  columnReorder: true,
  columns: [{
      title: "Value > 0",
      field: "c1",
      sortBy: "c1",
      conditions: [{
        backgroundColor: "green",
        color: "white",
        expression: "[c1] > 0"
      }]
    },
    {
      title: "Multiple Conditions",
      field: "c2",
      sortBy: "c2",
      conditions: [{
        backgroundColor: "",
        color: "green",
        expression: "[c2] > 0"
      }, {
        backgroundColor: "",
        color: "red",
        expression: "[c2] < 0"
      }, {
        backgroundColor: "",
        color: "blue",
        expression: "[c2] == 0"
      }]
    },
    {
      title: "Range between -3 and 3",
      field: "c3",
      sortBy: "c3",
      conditions: [{
        backgroundColor: "lightskyblue",
        color: "",
        expression: "[c3] >= -3 && [c3] <= 3"
      }]
    }
  ],
  dataModel: {
    fields: ["c1", "c2", "c3"],
    format: "array",
    data: [
      [10, -10, 4],
      [2, 10, 1],
      [0, -50, 2],
      [-1, 0, 9],
      [-5, 123.23, 3],
      [40, 0, 5],
      [500, -509.1, -1],
      [1200, 20.9, -2],
      [20, -40.8, -8],
      [-70, 2.7, 0],
      [-40, 1.3, 0],
      [100, -0.1, 0]
    ]
  },
  extensions: [
    new tr.ConditionalColoringExtension()
  ]
};

var grid = document.getElementById("grid");
grid.config = configObj;

  height: 208px;
}

Blinking feature

For required scenarios, there are features to deal with real-time data. For example, stock market data changes all the time, and a user needs to know that the data has changed. The Blinking feature was created to deal with scenarios like this. See the Blinking's typedef section for more details on this feature.

  • HTML
  • JS
  • CSS
<emerald-grid id="grid">

</emerald-grid>
var conditionalColoringExt = new tr.ConditionalColoringExtension();

var DataGenerator = tr.DataGenerator;
DataGenerator.addFieldInfo("num1", { type: "number", min: -10, max: 11 });
DataGenerator.addFieldInfo("num2", { type: "number", min: -1, max: 11 });
DataGenerator.addFieldInfo("num3", { type: "number", min: -10, max: 3 });
DataGenerator.addFieldInfo("num4", { type: "number", min: -1, max: 3 });
DataGenerator.addFieldInfo("num5", { type: "number", min: -5000, max: 5001 });
DataGenerator.addFieldInfo("char", { type: "number", min: ("A").charCodeAt(0), max: ("Z").charCodeAt(0) + 1 });

var fields = ["num1", "num2", "num3", "num4", "num5", "char"];
var records = DataGenerator.generateRecords(fields, { numRows: 15 });
records.forEach(function(record) {
  record.char = String.fromCharCode(record.char);
});

var configObj = {
  columnReorder: true,
  columns: [{
      title: "Blink with Theme Color",
      field: "num1",
      sortBy: "num1",
      blinking: true
    },
    {
      title: "Blink on Top of Condtional Coloring",
      field: "num2",
      sortBy: "num2",
      blinking: true,
      conditions: [{
        backgroundColor: "green",
        color: "white",
        expression: "[num2] > 0"
      }, {
        backgroundColor: "red",
        color: "white",
        expression: "[num2] < 0"
      }]
    },
    {
      title: "Blinking Border",
      field: "num3",
      sortBy: "num3",
      blinking: {
        border: true
      },
      conditions: [{
        backgroundColor: "",
        color: "green",
        expression: "[num3] > 0"
      }, {
        backgroundColor: "",
        color: "red",
        expression: "[num3] < 0"
      }, {
        backgroundColor: "",
        color: "blue",
        expression: "[num3] == 0"
      }]
    },
    {
      title: "Custom Blinking Color",
      field: "num4",
      sortBy: "num4",
      blinking: {
        up: "#aaDD88",
        down: "#ff33cc",
        level: "#111111"
      },
      conditions: [{
        backgroundColor: "lightskyblue",
        color: "",
        expression: "[num4] >= -3 && [num4] <= 3"
      }]
    },
    {
      title: "Blinking Border with Custom Color",
      field: "num5",
      sortBy: "num5",
      blinking: {
        border: true,
        up: "blue",
        down: "#ffff00"
      },
      conditions: [{
        backgroundColor: "salmon",
        color: "white",
        expression: "[num2] > 0"
      }]
    },
    {
      title: "Blink Only Specific State",
      field: "char",
      sortBy: "char",
      blinking: {
        level: "#bbbbbb"
      },
      colorText: "num3" // Using eikon theme to color
    }
  ],
  dataModel: {
    data: records
  },
  extensions: [
    conditionalColoringExt
  ]
};

var grid = document.getElementById("grid");
grid.config = configObj;

var randBetween = tr.DataGenerator.randBetween;

setInterval(function() { // Slow Update for testing
  var dv = grid.api.getDataView();
  var record = DataGenerator.generateRecord(fields);
  record.char = String.fromCharCode(record.char);
  dv.setRowDataAt(0, record);
}, 2000);

setInterval(function() { // Fast Updates
  var record = DataGenerator.generateRecord(fields);
  record.char = String.fromCharCode(record.char);

  // remove some fields randomly
  for (var key in record) {
    if (randBetween(0, 10) > 3) {
      delete record[key];
    }
  }

  var dv = grid.api.getDataView();
  var rowCount = dv.getRowCount();
  var at = randBetween(1, rowCount);
  dv.setRowDataAt(at, record);
}, 200);

Type Definitions

typedef

Blinking

Available options describing `blinking` object. Additionally if specify blinking as string of field, blinking will base on the field.
If specified true, field from column definition will be used.
Type:
string | boolean | Object
Properties:
Name Type Attributes Default Description
border boolean <optional>
 false If enabled, border will be blinked instead of background
field string <optional>
 If not specified, field from column definition will be used
up string <optional>
 CSS color (e.g. #00ff00, green)
down string <optional>
 CSS color (e.g. #ff0000, red)
level string <optional>
 CSS color (e.g. #33333, grey)

 null Alias of colorText.
blinking ConditionalColoringPlugin~Blinking <optional>
 null Blink Options. If specified, the cell will be blinked on data change.
field string <optional>
