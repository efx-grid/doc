Auto Tooltip

The Auto Tooltip Extension displays clipped information when the user hovers the pointer over the content or title.

For contributions, check out the Auto Tooltip Extension Repository.

  • HTML
  • JS
  • CSS
<emerald-grid id="grid">

</emerald-grid>
var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
var records = tr.DataGenerator.generateRecords(fields, { numRows: 5 });
var configObj = {
  rowHighlight: true,
  rowSelection: true,
  columns: [
    { title: "Company", field: fields[0] },
    { title: "Market", field: fields[1], width: 120 },
    { title: "Last", field: fields[2], width: 100 },
    { title: "Net. Chng", field: fields[3], width: 100 },
    { title: "Industry", field: fields[4], width: 150 }
  ],
  dataModel: {
    data: records
  },
  autoTooltip: {
    title: true
  },
  extensions: [
    new tr.AutoTooltipExtension()
  ]
};

var grid = document.getElementById("grid");
grid.config = configObj;
emerald-grid {
  height: 200px;
}

Note: the auto tooltip will not be added to non-clipped text.

Installation and importing to your project

Installation details and examples, along with information on how to import the extension to your project, are available on the Overview page.

Specific column

To enable Auto Tooltip in a specific column only:

  • Set content: false in the extension's option
  • Set autoTooltip: true in the column's option
  • HTML
  • JS
  • CSS
<emerald-grid id="grid">

</emerald-grid>
var fields = ["companyName", "market", "CF_LAST", "CF_NETCHNG", "industry"];
var records = tr.DataGenerator.generateRecords(fields, { numRows: 5 });
var configObj = {
  rowHighlight: true,
  rowSelection: true,
  columns: [
    { title: "Company", field: fields[0] },
    { title: "Market", field: fields[1], width: 120 },
    { title: "Last", field: fields[2], width: 100 },
    { title: "Net. Chng", field: fields[3], width: 100 },
    { title: "Industry (On)", field: fields[4], width: 150, autoTooltip: true },
    { title: "Industry (Off)", field: fields[4], width: 150 }
  ],
  dataModel: {
    data: records
  },
  autoTooltip: {
    title: true,
    content: false
  },
  extensions: [
    new tr.AutoTooltipExtension()
  ]
};

var grid = document.getElementById("grid");
grid.config = configObj;
emerald-grid {
  height: 200px;
}
var grid = document.getElementById('grid');
grid.config = {
  // any other grid's options
  columns: [
    { title: 'Column 1', field: 'field1', autoTooltip: true }
  ],
  autoTooltip: {
    title: true,
    content: false
  },
  extensions: [
    new tr.AutoTooltipExtension()
  ]
};

Limitation

Since Safari automatically apply tooltip to clipped text by itself. Therefore, Auto tooltip will be disabled in Safari.
