Heat Map

The Heat Map Extension provides heat map coloring for column fields, which will be the type set by the heatMap property.

efx-grid {
    height: 200px;
}

<efx-grid id="grid"></efx-grid>

await (await import("./theme-loader.js")).default; // This line is only required for demo purpose. It is not relevant for using EFX Grid in your application.

/* ---------------------------------- Note ----------------------------------
  DataGenerator, Formatters and extensions are exposed to global scope
  in the bundle file to make it easier to create live examples.
  Importing formatters and extensions is still required in your application. 
  Please see the document for further information.
---------------------------------------------------------------------------*/
DataGenerator.addFieldInfo("volume", {
    type: "number",
    min: "1",
    max: "1001"
});
var fields = ["companyName", "market", "CF_LAST", "volume", "PCTCHNG2"];
var records = DataGenerator.generateRecords(fields, { numRows: 20 });
var configObj = {
    columns: [
        {
            title: "Company",
            field: fields[0]
        },
        {
            title: "Market",
            field: fields[1],
            width: 120
        },
        {
            title: "Last",
            field: fields[2],
            width: 100,
            heatMap: {
                mode: "text",
                midPoint: 500
            }
        },
        {
            title: "Volume",
            field: fields[3],
            sortBy: fields[3],
            width: 100,
            heatMap: {
                midPoint: 300
            }
        },
        {

            title: "Pct. Chng",
            field: fields[4],
            sortBy: fields[4],
            width: 100,
            heatMap: true
        }
    ],
    staticDataRows: records,
    extensions: [
        new HeatMap()
    ]
};

var grid = document.getElementById("grid");
grid.config = configObj;
